DML for insert data to postgresql:

    #Library
    from sqlalchemy import create_engine
    import pandas as pd
    import codecs

    #Read CSV
    with codecs.open('Supply_Chain_Dataset.csv', 'r', encoding='ISO-8859-1') as f:
        df = pd.read_csv(f)

    #Create connection to postgresql
    engine = create_engine('postgresql://airflow:airflow@localhost:6607/airflow')

    #store dataframe to postgresql
    df.to_sql('supply_chain', engine, if_exists='replace', index=False)
